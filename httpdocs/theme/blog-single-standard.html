Google Now is an intelligent personal assistant developed by Google that is available within the Google Search mobile application for the Android and iOS operating systems. Google Now uses a natural language user interface to answer questions, make recommendations, and perform actions by delegating requests to a set of web services. 

(function(doc, script) {
  var js,

    frag = doc.createDocumentFragment(),
    add = function(url, id) {
      if (doc.getElementById(id)) {return;}
      js = doc.createElement(script);
      js.src = url;
      id && (js.id = id);
